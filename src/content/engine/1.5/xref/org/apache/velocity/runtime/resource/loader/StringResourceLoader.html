37  /**
38   * Resource loader that works with Strings. Users should manually add
39   * resources to the repository that is know by the factory of this package.
40   *
41   * Below is an example configuration for this loader.
42   * Note that 'repositoryimpl' is not mandatory;
43   * if not provided, the factory will fall back on using the default
44   * implementation of this package.
45   *
46   * string.resource.loader.description = Velocity StringResource loader
47   * string.resource.loader.class = org.apache.velocity.runtime.resource.loader..StringResourceLoader
48   * string.resource.loader.repository.class = org.apache.velocity.runtime.resource.loader.StringResourceRepositoryImpl
49   *
50   * Resources can be added to the repository like this:
51   * <code>
52   *   StringResourceRepository = StringResourceLoader.getRepository();
53   *
54   *   String myTemplateName = "/somewhere/intherepo/name";
55   *   String myTemplateBody = "Hi, ${username}... this is a some template!";
56   *   vsRepository.putStringResource(myTemplateName, myTemplateBody);
57   * </code>
58   *
59   * After this, the templates can be retrieved as usual.
60   *
61   * @author <a href="mailto:eelco.hillenius@openedge.nl">Eelco Hillenius</a>
62   * @author <a href="mailto:henning@apache.org">Henning P. Schmiedehausen</a>
63   * @version $Id: StringResourceLoader.java 479058 2006-11-25 00:26:32Z henning $
64   */
65  public class StringResourceLoader extends ResourceLoader
66  {
67      /** Key to look up the repository implementation class. */
68      public static final String REPOSITORY_CLASS = "repository.class";
69  
70      /** The default implementation class. */
71      public static final String REPOSITORY_CLASS_DEFAULT = StringResourceRepositoryImpl.class.getName();
72  
73      /** Key to look up the repository char encoding. */
74      public static final String REPOSITORY_ENCODING = "repository.encoding";
75  
76      /** The default repository encoding. */
77      public static final String REPOSITORY_ENCODING_DEFAULT = "UTF-8";
78  
79      /**
80       * Returns a reference to the Repository.
81       *
82       * @return A StringResourceRepository Reference.
83       */
84      public static StringResourceRepository getRepository()
85      {
86          return RepositoryFactory.getRepository();
87      }

89      /**
90       * @see org.apache.velocity.runtime.resource.loader.ResourceLoader#init(org.apache.commons.collections.ExtendedProperties)
91       */
92      public void init(final ExtendedProperties configuration)
93      {
94          log.info("StringResourceLoader : initialization starting.");
95  
96          String repositoryClass = configuration.getString(REPOSITORY_CLASS, REPOSITORY_CLASS_DEFAULT);
97          String encoding = configuration.getString(REPOSITORY_ENCODING, REPOSITORY_ENCODING_DEFAULT);
98  
99          RepositoryFactory.setRepositoryClass(repositoryClass);
100         RepositoryFactory.setEncoding(encoding);
101         RepositoryFactory.init(log);
102 
103         log.info("StringResourceLoader : initialization complete.");
104     }

106     /**
107      * Get an InputStream so that the Runtime can build a
108      * template with it.
109      *
110      * @param name name of template to get.
111      * @return InputStream containing the template.
112      * @throws ResourceNotFoundException Ff template not found
113      *         in the RepositoryFactory.
114      */
115     public InputStream getResourceStream(final String name)
116             throws ResourceNotFoundException
117     {
118         if (StringUtils.isEmpty(name))
119         {
120             throw new ResourceNotFoundException("No template name provided");
121         }
122 
123         StringResource resource = getRepository().getStringResource(name);
124         
125         if(resource == null)
126         {
127             throw new ResourceNotFoundException("Could not locate resource '" + name + "'");
128         }
129         
130         byte [] byteArray = null;
131     	
132         try
133         {
134             byteArray = resource.getBody().getBytes(resource.getEncoding());
135             return new ByteArrayInputStream(byteArray);
136         }
137         catch(UnsupportedEncodingException ue)
138         {
139             throw new VelocityException("Could not convert String using encoding " + resource.getEncoding(), ue);
140         }
141     }

143     /**
144      * @see org.apache.velocity.runtime.resource.loader.ResourceLoader#isSourceModified(org.apache.velocity.runtime.resource.Resource)
145      */
146     public boolean isSourceModified(final Resource resource)
147     {
148         StringResource original = null;
149         boolean result = true;
150 
151         original = getRepository().getStringResource(resource.getName());
152 
153         if (original != null)
154         {
155             result =  original.getLastModified() != resource.getLastModified();
156         }
157 
158         return result;
159     }
160 
161     /**
162      * @see org.apache.velocity.runtime.resource.loader.ResourceLoader#getLastModified(org.apache.velocity.runtime.resource.Resource)
163      */
164     public long getLastModified(final Resource resource)
165     {
166         StringResource original = null;
167 
168         original = getRepository().getStringResource(resource.getName());
169 
170         return (original != null)
171                 ? original.getLastModified()
172                 : 0;
173     }

176     /**
177      * Factory for constructing and obtaining the instance of
178      * StringResourceRepository implementation.
179      *
180      * Users can provide their own implementation by setting the property 'repository.class'
181      * for the resource loader. Note that at this time only one instance of a
182      * string resource repository can be used in a single VM.
183      *
184      * @author <a href="mailto:eelco.hillenius@openedge.nl">Eelco Hillenius</a>
185      * @author <a href="mailto:henning@apache.org">Henning P. Schmiedehausen</a>
186      * @version $Id: StringResourceLoader.java 479058 2006-11-25 00:26:32Z henning $
187      */
188     private static final class RepositoryFactory
189     {
190         /**
191          * is the factory initialised properly?
192          */
193         private static boolean isInitialized = false;
194 
195         /**
196          * repository instance
197          */
198         private static StringResourceRepository repository = null;
199 
200         /**
201          * Sets the repository class.
202          *
203          * @param className class that implements StringResourceRepository.
204          */
205         public static void setRepositoryClass(final String className)
206         {
207             if (isInitialized)
208             {
209                 throw new IllegalStateException("The RepositoryFactory has already been initialized!");
210             }
211 
212             try
213             {
214                 repository = (StringResourceRepository) ClassUtils.getNewInstance(className);
215             }
216             catch (ClassNotFoundException cnfe)
217             {
218                 throw new VelocityException("Could not find '" + className + "'", cnfe);
219             }
220             catch (IllegalAccessException iae)
221             {
222                 throw new VelocityException("Could not access '" + className + "'", iae);
223             }
224             catch (InstantiationException ie)
225             {
226                 throw new VelocityException("Could not instantiante '" + className + "'", ie);
227             }
228         }
229 
230         /**
231          * Sets the current repository encoding.
232          *
233          * @param encoding The current repository encoding.
234          */
235         public static void setEncoding(final String encoding)
236         {
237             if (repository == null)
238             {
239                 throw new IllegalStateException("The Repository class has not yet been set!");
240             }
241 
242             repository.setEncoding(encoding);
243         }
244 
245         /**
246          * Init the factory with the user given class name.
247          *
248          * @throws VelocityException If something goes wrong.
249          */
250         public static synchronized void init(final Log log)
251                 throws VelocityException
252         {
253             if (isInitialized)
254             {
255                 throw new IllegalStateException("Attempted to re-initialize Factory!");
256             }
257 
258             if (log.isInfoEnabled())
259             {
260                 log.info("Using " + repository.getClass().getName() + " as repository implementation");
261                 log.info("Current repository encoding is " + repository.getEncoding());
262             }
263             isInitialized = true;
264         }
265 
266         /**
267          * Get a reference to the repository.
268          * @return A StringResourceRepository implementation object.
269          */
270         public static StringResourceRepository getRepository()
271         {
272             if(!isInitialized)
273             {
274                 throw new IllegalStateException(
275                         "RepositoryFactory was not properly set up");
276             }
277             return repository;
278         }
279     }
