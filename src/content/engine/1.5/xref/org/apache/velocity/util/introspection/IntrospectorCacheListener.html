22  /**
23   * Interface to implement for listeners to the internal Introspector cache.
24   *
25   * @author <a href="mailto:henning@apache.org">Henning P. Schmiedehausen</a>
26   * @version $Id: IntrospectorCacheListener.java 476951 2006-11-19 22:22:43Z henning $
27   */
28  public interface IntrospectorCacheListener
29  {
30      /**
31       * Gets called when the Cache is cleared.
32       */
33      void triggerClear();
34  
35      /**
36       * Gets called when a ClassMap is requested from the Cache.
37       *
38       * @param c The class object to look up.
39       * @param classMap The Class map to return. Might be null.
40       */
41      void triggerGet(Class c, ClassMap classMap);
42  
43      /**
44       * Gets called when a ClassMap is put into the Cache.
45       *
46       * @param c The class object to look up.
47       * @param classMap The Class map stored in the Cache. Is never null.
48       */
49      void triggerPut(Class c, ClassMap classMap);
50  }
