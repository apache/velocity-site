95      /**
96       *  To support iterative objects used in a <code>#foreach()</code>
97       *  loop.
98       *
99       * @param obj The iterative object.
100      * @param i Info about the object's location.
101      * @return An {@link Iterator} object.
102      * @throws Exception
103      */
104     public Iterator getIterator(Object obj, Info i)
105             throws Exception
106     {
107         if (obj.getClass().isArray())
108         {
109             return new ArrayIterator(obj);
110         }
111         else if (obj instanceof Collection)
112         {
113             return ((Collection) obj).iterator();
114         }
115         else if (obj instanceof Map)
116         {
117             return ((Map) obj).values().iterator();
118         }
119         else if (obj instanceof Iterator)
120         {
121             if (log.isDebugEnabled())
122             {
123                 log.debug("The iterative object in the #foreach() loop at " +
124                            i + " is of type java.util.Iterator.  Because " +
125                            "it is not resettable, if used in more than once it " +
126                            "may lead to unexpected results.");
127             }
128             return ((Iterator) obj);
129         }
130         else if (obj instanceof Enumeration)
131         {
132             if (log.isDebugEnabled())
133             {
134                 log.debug("The iterative object in the #foreach() loop at " +
135                            i + " is of type java.util.Enumeration.  Because " +
136                            "it is not resettable, if used in more than once it " +
137                            "may lead to unexpected results.");
138             }
139             return new EnumerationIterator((Enumeration) obj);
140         }
141 
142         /*  we have no clue what this is  */
143         log.info("Could not determine type of iterator in #foreach loop at " + i);
144 
145         return null;
146     }

148     /**
149      *  Method
150      * @param obj
151      * @param methodName
152      * @param args
153      * @param i
154      * @return A Velocity Method.
155      * @throws Exception
156      */
157     public VelMethod getMethod(Object obj, String methodName, Object[] args, Info i)
158             throws Exception
159     {
160         if (obj == null)
161         {
162             return null;
163         }
164 
165         Method m = introspector.getMethod(obj.getClass(), methodName, args);
166 
167         return (m != null) ? new VelMethodImpl(m) : null;
168     }

170     /**
171      * Property  getter
172      * @param obj
173      * @param identifier
174      * @param i
175      * @return A Velocity Getter Method.
176      * @throws Exception
177      */
178     public VelPropertyGet getPropertyGet(Object obj, String identifier, Info i)
179             throws Exception
180     {
181         if (obj == null)
182         {
183             return null;
184         }
185 
186         Class claz = obj.getClass();
187 
188         /*
189          *  first try for a getFoo() type of property
190          *  (also getfoo() )
191          */
192         AbstractExecutor executor = new PropertyExecutor(log, introspector, claz, identifier);
193 
194         /*
195          * Let's see if we are a map...
196          */
197         if (!executor.isAlive()) 
198         {
199             executor = new MapGetExecutor(log, claz, identifier);
200         }
201 
202         /*
203          *  if that didn't work, look for get("foo")
204          */
205 
206         if (!executor.isAlive())
207         {
208             executor = new GetExecutor(log, introspector, claz, identifier);
209         }
210 
211         /*
212          *  finally, look for boolean isFoo()
213          */
214 
215         if (!executor.isAlive())
216         {
217             executor = new BooleanPropertyExecutor(log, introspector, claz,
218                                                    identifier);
219         }
220 
221         return (executor.isAlive()) ? new VelGetterImpl(executor) : null;
222     }

224     /**
225      * Property setter
226      * @param obj
227      * @param identifier
228      * @param arg
229      * @param i
230      * @return A Velocity Setter method.
231      * @throws Exception
232      */
233     public VelPropertySet getPropertySet(Object obj, String identifier,
234                                          Object arg, Info i)
235             throws Exception
236     {
237         if (obj == null)
238         {
239             return null;
240         }
241 
242         Class claz = obj.getClass();
243 
244         /*
245          *  first try for a setFoo() type of property
246          *  (also setfoo() )
247          */
248         SetExecutor executor = new SetPropertyExecutor(log, introspector, claz, identifier, arg);
249 
250         /*
251          * Let's see if we are a map...
252          */
253         if (!executor.isAlive())  {
254             executor = new MapSetExecutor(log, claz, identifier);
255         }
256 
257         /*
258          *  if that didn't work, look for put("foo", arg)
259          */
260 
261         if (!executor.isAlive())
262         {
263             executor = new PutExecutor(log, introspector, claz, arg, identifier);
264         }
265 
266         return (executor.isAlive()) ? new VelSetterImpl(executor) : null;
267     }
