22  import org.apache.velocity.runtime.RuntimeServices;
23  import org.apache.velocity.util.ClassUtils;
24  import org.apache.velocity.util.StringUtils;
25  
26  /**
27   * Factory to grab a template loader.
28   *
29   * @author <a href="mailto:jvanzyl@apache.org">Jason van Zyl</a>
30   * @version $Id: ResourceLoaderFactory.java 463298 2006-10-12 16:10:32Z henning $
31   */
32  public class ResourceLoaderFactory
33  {
34      /**
35       * Gets the loader specified in the configuration file.
36       * @param rs
37       * @param loaderClassName
38       * @return TemplateLoader
39       * @throws Exception
40       */
41      public static ResourceLoader getLoader(RuntimeServices rs, String loaderClassName)
42       throws Exception
43      {
44          ResourceLoader loader = null;
45  
46          try
47          {
48              loader = (ResourceLoader) ClassUtils.getNewInstance( loaderClassName );
49  
50              rs.getLog().debug("ResourceLoader instantiated: "
51                                + loader.getClass().getName());
52  
53              return loader;
54          }
55          // The ugly three strike again: ClassNotFoundException,IllegalAccessException,InstantiationException
56          catch(Exception e)
57          {
58              rs.getLog().error("Problem instantiating the template loader.\n" +
59                            "Look at your properties file and make sure the\n" +
60                            "name of the template loader is correct. Here is the\n" +
61                            "error:", e);
62  
63              throw new Exception("Problem initializing template loader: " + loaderClassName +
64              "\nError is: " + StringUtils.stackTrace(e));
65          }
66      }
67  }
