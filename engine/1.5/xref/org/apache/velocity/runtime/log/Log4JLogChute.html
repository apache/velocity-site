32  /**
33   * Implementation of a simple log4j system that will either latch onto
34   * an existing category, or just do a simple rolling file log.
35   *
36   * @author <a href="mailto:geirm@apache.org>Geir Magnusson Jr.</a>
37   * @author <a href="mailto:dlr@finemaltcoding.com>Daniel L. Rall</a>
38   * @author <a href="mailto:nbubna@apache.org>Nathan Bubna</a>
39   * @version $Id: Log4JLogChute.java 463298 2006-10-12 16:10:32Z henning $
40   * @since Velocity 1.5
41   */
42  public class Log4JLogChute implements LogChute
43  {
44      /**
45       *
46       */
47      public static final String RUNTIME_LOG_LOG4J_LOGGER =
48              "runtime.log.logsystem.log4j.logger";
49  
50      private RuntimeServices rsvc = null;
51      private boolean hasTrace = false;
52      private RollingFileAppender appender = null;
53  
54      /**
55       * <a href="http://jakarta.apache.org/log4j/">Log4J</a> logging API.
56       */
57      protected Logger logger = null;
58  
59      /**
60       * @see org.apache.velocity.runtime.log.LogChute#init(org.apache.velocity.runtime.RuntimeServices)
61       */
62      public void init(RuntimeServices rs) throws Exception
63      {
64          rsvc = rs;
65  
66          /* first see if there is a category specified and just use that - it allows
67           * the application to make us use an existing logger
68           */
69          String name = (String)rsvc.getProperty(RUNTIME_LOG_LOG4J_LOGGER);
70          if (name != null)
71          {
72              logger = Logger.getLogger(name);
73              log(DEBUG_ID, "Log4JLogChute using logger '" + name + '\'');
74          }
75          else
76          {
77              // create a logger with this class name to avoid conflicts
78              logger = Logger.getLogger(this.getClass().getName());
79  
80              // if we have a file property, then create a separate
81              // rolling file log for velocity messages only
82              String file = rsvc.getString(RuntimeConstants.RUNTIME_LOG);
83              if (file != null && file.length() > 0)
84              {
85                  initAppender(file);
86              }
87          }
88  
89          /* Ok, now let's see if this version of log4j supports the trace level. */
90          try
91          {
92              Field traceLevel = Level.class.getField("TRACE");
93              // we'll never get here in pre 1.2.12 log4j
94              hasTrace = true;
95          }
96          catch (NoSuchFieldException e)
97          {
98              log(DEBUG_ID,
99                  "The version of log4j being used does not support the \"trace\" level.");
100         }
101     }
102 
103     // This tries to create a file appender for the specified file name.
104     private void initAppender(String file) throws Exception
105     {
106         try
107         {
108             // to add the appender
109             PatternLayout layout = new PatternLayout("%d - %m%n");
110             this.appender = new RollingFileAppender(layout, file, true);
111 
112             // if we successfully created the file appender,
113             // configure it and set the logger to use only it
114             appender.setMaxBackupIndex(1);
115             appender.setMaximumFileSize(100000);
116 
117             // don't inherit appenders from higher in the logger heirarchy
118             logger.setAdditivity(false);
119             // this impl checks levels (by default don't include trace level)
120             logger.setLevel(Level.DEBUG);
121             logger.addAppender(appender);
122             log(DEBUG_ID, "Log4JLogChute initialized using file '"+file+'\'');
123         }
124         catch (IOException ioe)
125         {
126             rsvc.getLog().warn("Could not create file appender '"+file+'\'', ioe);
127             throw ExceptionUtils.createRuntimeException("Error configuring Log4JLogChute : ", ioe);
128         }
129     }

131     /**
132      *  logs messages
133      *
134      *  @param level severity level
135      *  @param message complete error message
136      */
137     public void log(int level, String message)
138     {
139         switch (level)
140         {
141             case LogChute.WARN_ID:
142                 logger.warn(message);
143                 break;
144             case LogChute.INFO_ID:
145                 logger.info(message);
146                 break;
147             case LogChute.DEBUG_ID:
148                 logger.debug(message);
149                 break;
150             case LogChute.TRACE_ID:
151                 if (hasTrace)
152                 {
153                     logger.trace(message);
154                 }
155                 else
156                 {
157                     logger.debug(message);
158                 }
159                 break;
160             case LogChute.ERROR_ID:
161                 logger.error(message);
162                 break;
163             default:
164                 logger.debug(message);
165                 break;
166         }
167     }

169     /**
170      * @see org.apache.velocity.runtime.log.LogChute#log(int, java.lang.String, java.lang.Throwable)
171      */
172     public void log(int level, String message, Throwable t)
173     {
174         switch (level)
175         {
176             case LogChute.WARN_ID:
177                 logger.warn(message, t);
178                 break;
179             case LogChute.INFO_ID:
180                 logger.info(message, t);
181                 break;
182             case LogChute.DEBUG_ID:
183                 logger.debug(message, t);
184                 break;
185             case LogChute.TRACE_ID:
186                 if (hasTrace)
187                 {
188                     logger.trace(message, t);
189                 }
190                 else
191                 {
192                     logger.debug(message, t);
193                 }
194                 break;
195             case LogChute.ERROR_ID:
196                 logger.error(message, t);
197                 break;
198             default:
199                 logger.debug(message, t);
200                 break;
201         }
202     }

204     /**
205      * @see org.apache.velocity.runtime.log.LogChute#isLevelEnabled(int)
206      */
207     public boolean isLevelEnabled(int level)
208     {
209         switch (level)
210         {
211             case LogChute.DEBUG_ID:
212                 return logger.isDebugEnabled();
213             case LogChute.INFO_ID:
214                 return logger.isInfoEnabled();
215             case LogChute.TRACE_ID:
216                 if (hasTrace)
217                 {
218                     return logger.isTraceEnabled();
219                 }
220                 else
221                 {
222                     return logger.isDebugEnabled();
223                 }
224             case LogChute.WARN_ID:
225                 return logger.isEnabledFor(Level.WARN);
226             case LogChute.ERROR_ID:
227                 // can't be disabled in log4j
228                 return logger.isEnabledFor(Level.ERROR);
229             default:
230                 return true;
231         }
232     }

234     /**
235      * Also do a shutdown if the object is destroy()'d.
236      * @throws Throwable
237      */
238     protected void finalize() throws Throwable
239     {
240         shutdown();
241     }
242 
243     /** Close all destinations*/
244     public void shutdown()
245     {
246         if (appender != null)
247         {
248             logger.removeAppender(appender);
249             appender.close();
250             appender = null;
251         }
252     }
